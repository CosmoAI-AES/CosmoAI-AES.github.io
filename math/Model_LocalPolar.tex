% Model_LocalPolar.tex
% Face-on view of the lens plane, showing the local polar coordinate
% system (theta-tilde, phi) centred on the image, together with the
% global coordinate directions (theta_1, theta_2).
% Colour palette chosen at mid luminance so it stays legible on both
% white and black page backgrounds.
\documentclass[border=4pt]{standalone}
\usepackage{tikz}
\usepackage{amsmath}
\usetikzlibrary{arrows.meta, decorations.markings, calc}

\definecolor{inkcol}{RGB}{125,132,140}
\definecolor{planecol}{RGB}{155,155,155}
\definecolor{lenscol}{RGB}{215,135,45}
\definecolor{imgcol}{RGB}{205,165,45}
\definecolor{anglecol}{RGB}{170,95,200}

% the kidney/banana shaped blob used for the distorted lensed image.
% Shifted so that the pic's local origin (0,0) -- which is where the
% polar coordinate system below is centred -- falls inside the filled
% shape, roughly at its "light centre", rather than in the concave notch.
\tikzset{
  kidney/.pic={
    \fill[imgcol]
      (-0.6,1) .. controls (-0.05,0.95) and (0.4,0.4)  .. (0.4,0)
               .. controls (0.4,-0.4)  and (-0.05,-0.95) .. (-0.6,-1)
               .. controls (-0.45,-0.75) and (-0.25,-0.3) .. (-0.25,0)
               .. controls (-0.25,0.3)  and (-0.45,0.75) .. cycle;
  }
}

\begin{document}
\begin{tikzpicture}[
    >=Latex,
    line width=1pt,
    font=\sffamily
  ]

  % --- coordinates -------------------------------------------------------
  \coordinate (Origin) at (0,0);
  \coordinate (ImgPos) at (4.2,2.4);

  % --- frame, representing the lens plane viewed face-on -----------------
  \draw[planecol, thin] (-1.1,-1.1) rectangle (7.1,6.1);
  \node[inkcol, font=\sffamily\bfseries, anchor=north west] at (-1.05,6.05) {LENS PLANE (face-on)};

  % --- global coordinate axes, centred on the lens ------------------------
  \draw[inkcol, thick, -{Latex[length=3mm]}] (Origin) -- (6.3,0) node[right, font=\sffamily\large] {$\theta_1$};
  \draw[inkcol, thick, -{Latex[length=3mm]}] (Origin) -- (0,4.6) node[above, font=\sffamily\large] {$\theta_2$};
  \fill[lenscol] (Origin) circle (0.18);
  \node[lenscol, font=\sffamily\bfseries\large, anchor=north east] at (-0.15,-0.15) {LENS};

  % --- the image itself (kidney/banana shape), concave side facing the lens
  \pic[scale=1.3, rotate=29.74] at (ImgPos) {kidney};
  \node[inkcol, font=\sffamily\large, anchor=south] at ($(ImgPos)+(0,1.7)$) {IMAGE};

  % --- local polar axis (phi = 0 reference direction, parallel to theta_1)
  \draw[anglecol, dotted, thick] (ImgPos) -- ($(ImgPos)+(1.9,0)$);

  % --- radius theta-tilde: the tip lands inside the image itself ---------
  \draw[anglecol, thick, -{Latex[length=2mm]}]
    (ImgPos) -- ($(ImgPos)+(50:0.5)$);
  \node[anglecol, font=\sffamily\large, anchor=west] at ($(ImgPos)+(50:0.95)$) {$\tilde{\theta}$};

  % --- angle phi, measured from the local polar axis ----------------------
  \draw[anglecol, thick, -{Latex[length=1.5mm]}]
    ($(ImgPos)+(0:0.24)$) arc [start angle=0, end angle=50, radius=0.24];
  \node[anglecol, font=\sffamily\large] at ($(ImgPos)+(18:0.68)$) {$\phi$};

\end{tikzpicture}
\end{document}
