% Model_2Db.tex
% TikZ recreation of the flat-sky lensing set-up figure (replaces Model_2Db.jpeg)
% Colour palette is chosen at mid luminance so every element stays legible
% on both white and black page backgrounds.
\documentclass[border=4pt]{standalone}
\usepackage{tikz}
\usepackage{amsmath}
\usetikzlibrary{arrows.meta, decorations.markings, calc}

\definecolor{inkcol}{RGB}{125,132,140}
\definecolor{planecol}{RGB}{155,155,155}
\definecolor{lenscol}{RGB}{215,135,45}
\definecolor{imgcol}{RGB}{205,165,45}
\definecolor{veccol}{RGB}{215,70,70}
\definecolor{anglecol}{RGB}{170,95,200}
\definecolor{obscol}{RGB}{95,150,220}

\begin{document}
\begin{tikzpicture}[
    >=Latex,
    line width=1pt,
    font=\sffamily
  ]

  % --- coordinates -----------------------------------------------------
  \coordinate (Observer)    at (0,0);
  \coordinate (Lens)        at (0,3);
  \coordinate (LensImg)     at (2.6,3);
  \coordinate (Source)      at (2.1,8.6);
  \coordinate (SourceImg)   at (7.6,8.6);
  \coordinate (AxisTop)     at (0,11);

  % --- plane lines -------------------------------------------------------
  \draw[planecol, thick] (-1,3)   -- (10.4,3)   node[right, inkcol, font=\sffamily\bfseries] {LENS PLANE};
  \draw[planecol, thick] (-1,8.6) -- (10.4,8.6) node[right, inkcol, font=\sffamily\bfseries] {SOURCE PLANE};

  % --- optical axis --------------------------------------------------
  \draw[inkcol, thick, -{Latex[length=3mm]}] (Observer) -- (AxisTop);
  \node[inkcol, rotate=90, font=\sffamily\bfseries\Large] at (-0.55,5.5) {OPTICAL AXIS};

  % --- distance measures (D_S, D_L) -----------------------------------
  \draw[planecol, {Latex[length=2mm]}-{Latex[length=2mm]}] (-1.4,0) -- (-1.4,8.6);
  \node[inkcol, font=\sffamily\bfseries] at (-1.75,4.3) {$D_{\mathrm S}$};

  \draw[planecol, {Latex[length=2mm]}-{Latex[length=2mm]}] (-0.9,0) -- (-0.9,3);
  \node[inkcol, font=\sffamily\bfseries] at (-0.55,1.5) {$D_{\mathrm L}$};

  % --- light rays (bent at the lens plane) ----------------------------
  \draw[lenscol, thick] (Observer) -- (LensImg) -- (SourceImg);
  \draw[lenscol, thick, dashed] (Source) -- (LensImg);

  % deflection angle marker: arc from the optical axis to the ray, at the observer
  \draw[anglecol, thick, -{Latex[length=2mm]}]
    ({0.7*cos(90)},{0.7*sin(90)}) arc [start angle=90, end angle=49.09, radius=0.7];
  \node[anglecol, font=\sffamily] at ({1.05*cos(69.5)},{1.05*sin(69.5)}) {$\boldsymbol{\theta}$};

  % --- vectors in the two planes --------------------------------------
  \draw[veccol, thick, -{Latex[length=3mm]}] (Lens) -- (LensImg)
    node[midway, above, font=\sffamily] {$\vec{\xi}$};

  \draw[veccol, thick, -{Latex[length=3mm]}] (0,8.6) -- (SourceImg);
  \node[veccol, font=\sffamily] at (1.05,8.9) {$\vec{\eta}_{\mathrm{ACT}}$};
  \node[veccol, font=\sffamily] at (5.0,8.9)  {$\Delta\vec{\eta}$};

  % --- lens, source, and image markers --------------------------------
  \fill[lenscol] (Lens) circle (0.22);
  \node[lenscol, font=\sffamily\bfseries, anchor=south] at ($(Lens)+(0,0.3)$) {LENS};

  \fill[imgcol] (LensImg) circle (0.2);
  \node[inkcol, font=\sffamily, anchor=south] at ($(LensImg)+(0,0.3)$) {IMAGE};

  \fill[imgcol] (SourceImg) circle (0.22);
  \node[inkcol, font=\sffamily, anchor=south] at ($(SourceImg)+(0,0.3)$) {IMAGE};

  \node[veccol, font=\sffamily] at (Source) {$\times$};
  \node[inkcol, font=\sffamily\bfseries, anchor=south] at ($(Source)+(0,0.35)$) {SOURCE};

  % --- observer icon ---------------------------------------------------
  \draw[obscol, thick] (Observer) circle (0.28);
  \node[obscol, font=\sffamily] at (0.75,-0.55) {observer};

\end{tikzpicture}
\end{document}
