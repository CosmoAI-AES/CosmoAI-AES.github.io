% Model_2Db.tex
% TikZ recreation of the flat-sky lensing set-up figure (replaces Model_2Db.jpeg)
% Colour palette is chosen at mid luminance so every element stays legible
% on both white and black page backgrounds.
\documentclass[border=4pt]{standalone}
\usepackage{tikz}
\usepackage{amsmath}
\usetikzlibrary{arrows.meta, decorations.markings, calc}

\definecolor{inkcol}{RGB}{125,132,140}
\definecolor{planecol}{RGB}{155,155,155}
\definecolor{lenscol}{RGB}{215,135,45}
\definecolor{imgcol}{RGB}{205,165,45}
\definecolor{anglecol}{RGB}{170,95,200}
\definecolor{obscol}{RGB}{95,150,220}

% a kidney/banana shaped blob used for the distorted lensed images,
% drawn as a single closed outline (convex outer edge, concave inner edge)
\tikzset{
  kidney/.pic={
    \fill[imgcol]
      (0,1) .. controls (0.55,0.95) and (1.0,0.4)  .. (1.0,0)
            .. controls (1.0,-0.4)  and (0.55,-0.95) .. (0,-1)
            .. controls (0.15,-0.75) and (0.35,-0.3) .. (0.35,0)
            .. controls (0.35,0.3)  and (0.15,0.75) .. cycle;
  }
}

\begin{document}
\begin{tikzpicture}[
    >=Latex,
    line width=1pt,
    font=\sffamily
  ]

  % --- coordinates -----------------------------------------------------
  \coordinate (Observer)    at (0,0);
  \coordinate (Lens)        at (0,3);
  \coordinate (LensImg)     at (2.6,3);
  \coordinate (Source)      at (2.1,8.6);
  \coordinate (SourceImg)   at (7.6,8.6);
  \coordinate (AxisTop)     at (0,11);

  % angles (measured from the positive x-axis) of the two rays from the observer
  %   ray to the lens-plane image (bent ray)      : 49.09 deg  ->  theta = 90-49.09
  %   straight line to the true source position    : 76.26 deg  ->  beta  = 90-76.26

  % --- plane lines -------------------------------------------------------
  \draw[planecol, thick] (-1,3)   -- (10.4,3)   node[right, inkcol, font=\sffamily\bfseries] {LENS PLANE};
  \draw[planecol, thick] (-1,8.6) -- (10.4,8.6) node[right, inkcol, font=\sffamily\bfseries] {SOURCE PLANE};

  % --- optical axis --------------------------------------------------
  \draw[inkcol, thick, -{Latex[length=3mm]}] (Observer) -- (AxisTop);
  \node[inkcol, rotate=90, font=\sffamily\bfseries\Large] at (-0.55,5.5) {OPTICAL AXIS};

  % --- distance measures (D_S, D_L) -----------------------------------
  \draw[planecol, {Latex[length=2mm]}-{Latex[length=2mm]}] (-1.4,0) -- (-1.4,8.6);
  \node[inkcol, font=\sffamily\bfseries] at (-1.75,4.3) {$D_{\mathrm S}$};

  \draw[planecol, {Latex[length=2mm]}-{Latex[length=2mm]}] (-0.9,0) -- (-0.9,3);
  \node[inkcol, font=\sffamily\bfseries] at (-0.55,1.5) {$D_{\mathrm L}$};

  % --- light rays (bent at the lens plane) ----------------------------
  \draw[lenscol, thick] (Observer) -- (LensImg) -- (SourceImg);
  \draw[lenscol, thick, dashed] (Source) -- (LensImg);

  % true (unlensed) direction to the source, from the observer
  \draw[anglecol, thick, dotted] (Observer) -- (Source);

  % --- angular markers at the observer: beta (axis to true source direction),
  % alpha = theta-beta (true source direction to bent ray), theta (axis to bent ray) ---
  \draw[anglecol, thick, -{Latex[length=2mm]}]
    ({2.4*cos(90)},{2.4*sin(90)}) arc [start angle=90, end angle=76.26, radius=2.4];
  \node[anglecol, font=\sffamily] at ({2.75*cos(83.1)},{2.75*sin(83.1)}) {$\boldsymbol{\beta}$};

  % alpha = theta-beta: the angle between the true source direction and the bent ray
  \draw[anglecol, thick, -{Latex[length=2mm]}]
    ({1.0*cos(76.26)},{1.0*sin(76.26)}) arc [start angle=76.26, end angle=49.09, radius=1.0];
  \node[anglecol, font=\sffamily] at ({1.35*cos(62.7)},{1.35*sin(62.7)}) {$\boldsymbol{\alpha}$};

  \draw[anglecol, thick, -{Latex[length=2mm]}]
    ({1.7*cos(90)},{1.7*sin(90)}) arc [start angle=90, end angle=49.09, radius=1.7];
  \node[anglecol, font=\sffamily] at ({2.05*cos(69.5)},{2.05*sin(69.5)}) {$\boldsymbol{\theta}$};

  % --- lens, source, and image markers --------------------------------
  \fill[lenscol] (Lens) circle (0.22);
  \node[lenscol, font=\sffamily\bfseries, anchor=south] at ($(Lens)+(0,0.3)$) {LENS};

  \pic[scale=0.22] at (LensImg) {kidney};
  \node[inkcol, font=\sffamily, anchor=south] at ($(LensImg)+(0,0.32)$) {IMAGE};

  \pic[scale=0.24] at (SourceImg) {kidney};
  \node[inkcol, font=\sffamily, anchor=south] at ($(SourceImg)+(0,0.34)$) {IMAGE};

  \node[inkcol, font=\sffamily] at (Source) {$\times$};
  \node[inkcol, font=\sffamily\bfseries, anchor=south] at ($(Source)+(0,0.35)$) {SOURCE};

  % --- observer icon (a smiley face) ------------------------------------
  \draw[obscol, thick] (Observer) circle (0.28);
  \fill[obscol] ($(Observer)+(-0.09,0.08)$) circle (0.032);
  \fill[obscol] ($(Observer)+(0.09,0.08)$) circle (0.032);
  \draw[obscol, thick]
    ($(Observer)+(-0.13,-0.02)$)
      .. controls ($(Observer)+(-0.05,-0.16)$) and ($(Observer)+(0.05,-0.16)$) ..
    ($(Observer)+(0.13,-0.02)$);
  \node[obscol, font=\sffamily] at (0.75,-0.55) {observer};

\end{tikzpicture}
\end{document}
